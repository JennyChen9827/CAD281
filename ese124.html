    int d;

        case 1: printf("Monday\n"); break;
        case 2: printf("Tuesday\n"); break;
        case 3: printf("Wednesday\n"); break;
        case 4: printf("Thursday\n"); break;

}

Exercise: Write a function to find the maximum value in an array.
Solution 
int find_max(int a[], int n) {
    int mx = a[0];
    for (int i = 1; i < n; i++) {
        if (a[i] > mx) mx = a[i];
    }
    return mx;
}

int main() {
    int arr[] = {3, 7, 2, 9, 5};
    printf("Max = %d\n", find_max(arr, 5));
    return 0;
}

Exercise: Implement a recursive function to compute factorial(n).
Solution 
long factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

int main() {
    printf("5! = %ld\n", factorial(5));
    return 0;
}

Exercise: Write a function to swap two integers via pointers.
Solution 
swap(int *a, int *b) {
    int tmp = *a;
    *a = *b;
    *b = tmp;
}

int main() {
    int a = 5, b = 7;
    swap(&a, &b);
    printf("a=%d, b=%d\n", a, b);
    return 0;
}

Exercise: Read integers from input.txt until EOF and print their sum.
Solution 
#include <stdio.h>

int main() {
    FILE *f = fopen("input.txt","r");
    if (!f) return 1;
    int sum=0, x;
    while (fscanf(f,"%d",&x)==1) sum += x;
    fclose(f);
    printf("Sum = %d\n", sum);
    return 0;
}

Bubble Sort

bubble(int a[], int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - i - 1; j++)
            if (a[j] > a[j + 1]) {
                int tmp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = tmp;
            }
}
Common Errors:
  • Loop bounds off by one.
Exercise: Implement insertion sort.
Solution 
insertion(int a[], int n) {
    for (int i = 1; i < n; i++) {
        int key = a[i], j = i - 1;
        while (j >= 0 && a[j] > key) {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = key;
    }
}

get_in(char c){

int main() {

    int out[5][3] = {

    };
    printf("Input: "); scanf("%s",s);
    for(int i=0;s[i];i++){
        int in=get_in(s[i]);
        cnt += out[st][in];
        st = nxt[st][in];
    }

}

Bitwise Manipulation

unsigned a = 5;       // 0b0101
unsigned b = a << 1;  // 0b1010 == 10
unsigned c = a & 3;   // 0b0101 & 0b0011 == 1
printf("b=%u, c=%u\n", b, c);
Exercise: Write a function to check if the nᵗʰ bit of x is set.
Solution 
is_bit_set(unsigned x, int n) {
    return (x >> n) & 1;
}

int main() {
    unsigned v=13; // 0b1101
    for(int i=0;i<4;i++){
        printf("bit %d of %u: %d\n", i, v, is_bit_set(v,i));
    }
    return 0;
}
